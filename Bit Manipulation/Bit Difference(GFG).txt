class Solution{
    
    // Function to find number of bits needed to be flipped to convert A to B
    public static int countBitsFlip(int a, int b){
        
        // Your code here
        //      01010
        //      10100
        // XOR  11110
        // count number of set bits in XOR
        int XOR  = a ^ b;
        int count  = 0;
        while(XOR!=0)
        {
            XOR = XOR & (XOR - 1);
            count++;
        }
        return count;
    }
    
    
}
